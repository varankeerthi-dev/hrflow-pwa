rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to get user's orgId
    function getUserOrgId() {
      return request.auth != null ? get(/databases/$(database)/documents/users/$(request.auth.uid)).data.orgId : null;
    }
    
    // Helper function to check if user is admin
    function isAdmin() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    
    // Helper function to check if user is HR
    function isHR() {
      return request.auth != null && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'HR';
    }
    
    // Helper function to check if user is admin or HR
    function isAdminOrHR() {
      return isAdmin() || isHR();
    }
    
    // Helper function to check if user has a role (any role means they're an org member)
    function isOrgMember() {
      return request.auth != null && getUserOrgId() != null;
    }
    
    // Users collection - read/write own document
    match /users/{uid} {
      allow read, create, update: if request.auth != null && request.auth.uid == uid;
    }
    
    // Organisations collection
    match /organisations/{orgId} {
      allow get: if request.auth != null;
      allow list: if request.auth != null && getUserOrgId() == orgId;
      allow write: if (isAdminOrHR() && getUserOrgId() == orgId) || (request.auth != null && getUserOrgId() == null);
      
      // Roles - read by all org members, write by admin
      match /roles/{roleId} {
        allow read: if request.auth != null && getUserOrgId() == orgId;
        allow write, delete: if isAdmin() && getUserOrgId() == orgId;
      }
      
      // Employees - any org member can read, admin/HR can write
      match /employees/{empId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow write, delete: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      // Shifts
      match /shifts/{shiftId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow write: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      // ATTENDANCE - FIXED: Any org member can read/write, not just their own
      match /attendance/{date}/{empId} {
        // All org members can read/write attendance for any employee
        allow read, write: if isOrgMember() && getUserOrgId() == orgId;
        // Admin/HR can also delete
        allow delete: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      // All other collections follow same pattern
      match /corrections/{corrId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow create: if isOrgMember() && getUserOrgId() == orgId;
        allow update: if isAdminOrHR() && getUserOrgId() == orgId;
        allow delete: if isAdmin() && getUserOrgId() == orgId;
      }
      
      match /otApprovals/{approvalId} {
        allow read, create: if isOrgMember() && getUserOrgId() == orgId;
        allow update: if isAdminOrHR() && getUserOrgId() == orgId;
        allow delete: if isAdmin() && getUserOrgId() == orgId;
      }
      
      match /otRequests/{requestId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow create: if isOrgMember() && getUserOrgId() == orgId;
        allow update: if isAdminOrHR() && getUserOrgId() == orgId;
        allow delete: if isAdmin() && getUserOrgId() == orgId;
      }
      
      match /advances/{advanceId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow create: if isOrgMember() && getUserOrgId() == orgId;
        allow update, delete: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      match /leaves/{leaveId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow create: if isOrgMember() && getUserOrgId() == orgId;
        allow update, delete: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      match /fines/{fineId} {
        allow read: if isOrgMember() && getUserOrgId() == orgId;
        allow create: if isOrgMember() && getUserOrgId() == orgId;
        allow update, delete: if isAdminOrHR() && getUserOrgId() == orgId;
      }
      
      match /activityLogs/{logId} {
        allow read, create: if isOrgMember() && getUserOrgId() == orgId;
      }
      
      match /logs/{logId} {
        allow read, create: if isOrgMember() && getUserOrgId() == orgId;
      }
    }
    
    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
